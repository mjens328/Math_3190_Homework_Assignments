gghist <- function(data, variable, group_var = NULL) {
  col_data <- data |> pull({{ variable }})
  
  if (!is.numeric(col_data)) {
    message("Variable is NOT Numeric") 
  } else {
    cat("Choose Histogram Type: \n1 Horizontal \n2 Vertical\n")
    user_choice <- readline(prompt = "Enter choice (1 or 2): ")
    
    p <- ggplot(data, aes(x = {{ variable }})) +
      geom_histogram(color = "black", fill = "lightblue4", bins = 10) + 
      theme_minimal() + 
      labs(title = paste("Histogram of", rlang::as_label(enquo(variable))))
    
    if (!rlang::quo_is_null(enquo(group_var))) {
      p <- p + facet_wrap(vars({{ group_var }})) +
               aes(fill = {{ group_var }}) + 
               theme(legend.position = "none")
    }
    
    if (!(user_choice %in% c("1", "2"))) {
      return("Not a Valid Choice. Start Over") 
    } else {
      if (user_choice == "1") {
        p <- p + coord_flip()
      }
      print(p)
    }
  }
}
ggbox <- function(data, variable, group_var = NULL) {
  col_data <- data |> pull({{ variable }})
  
  if (!is.numeric(col_data)) {
    message("Variable is NOT Numeric") 
  } else {
    cat("Choose Layout: \n1 Horizontal \n2 Vertical\n")
    user_choice <- readline(prompt = "Enter choice (1 or 2): ")
    
    if (rlang::quo_is_null(enquo(group_var))) {
      p <- ggplot(data, aes(x = "Data", y = {{ variable }}))
    } else {
      p <- ggplot(data, aes(x = {{ group_var }}, y = {{ variable }}, fill = {{ group_var }}))
    }
    
    p <- p + 
      geom_boxplot(color = "black", fill = "lightblue4") + 
      theme_minimal() + 
      labs(title = paste("Boxplot of", rlang::as_label(enquo(variable))))
    
    if (!rlang::quo_is_null(enquo(group_var))) {
      p <- p + theme(legend.position = "none")
    }

    if (!(user_choice %in% c("1", "2"))) {
      return("Not a Valid Choice. Start Over") 
    } else {
      if (user_choice == "1") { p <- p + coord_flip() }
      print(p)
    }
  }
}

summary_stats <- function(team_name, year) {
  ccbga6 |>
    filter(year(mdy(Game_Date)) == year) |> 
    filter(Home_Team == team_name | Away_Team == team_name) |>
    mutate(Outcome = case_when(
      Home_Team == team_name & Score_Difference > 0 ~ "Win",
      Away_Team == team_name & Score_Difference < 0 ~ "Win",
      TRUE ~ "Loss"
    )) |>
    summarize(
      Wins = sum(Outcome == "Win"),
      Losses = sum(Outcome == "Loss"),
      Winning_Percentage = Wins / (Wins + Losses)
    )
}
